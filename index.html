Scroll View

Transform HTML element into scrollable view manually using it as library with external custom scrollbar and gesture events.


    var $scroll = document.getElementById("scroll-vertical");
    var options = {
        "className": "css-class-scrollbar-vertical",
        "direction": "vertical"
    };
    $scroll.scrollBar = new RADJS_ScrollView.ScrollBar($scroll, options);
    var scrollviewOptions = {
        "bounds": true
    };
    $scroll.scroller = new RADJS_ScrollView($scroll, scrollViewOptions);

    Use scrollable view as WebComponent.

    
    <div
        is="x-radjs-scrollview"
        options="
            direction: horizontal;
            bounds: false;
            scrollbar: css-class-scroll-horizontal;
            smart: #scroll-horizontal
        "
    >
        <div id="scroll-horizontal" class="scroll-content">
            Content…
        </div>
    </div>
